        function test_func(x,y) result(f)
        !! 2d test function

        implicit none

        real(wp) :: f
        real(wp),intent(in) :: x,y

        real(wp),parameter :: deg2rad = acos(-1.0_wp)/180.0_wp  !! degrees to radians conversion factor

        f = sin(deg2rad*(x+y))

        end function test_func
